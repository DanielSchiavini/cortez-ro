prontera,146,163,6	script	Novice Castles	729,{
	mes "[Cita]";
	mes "Hey! I'm a new usher of Novice Castles.";
	next;
	if( (Class>=Job_Novice && Class<=Job_Thief)
		|| (Class==Job_Taekwon && BaseLevel<90) || Class==Job_SuperNovice || Class==Job_Super_Baby
		|| (Class>=Job_Baby && Class<=Job_Baby_Thief)
		|| (Class>=Job_Novice_High && Class<=Job_Thief_High && BaseLevel<90)
	) menu "Warp me to Novice Castles",M_WARP,"Cancel",-;

	mes "[Cita]";
	mes "All the 2nd classes aren't allowed to enter the sacred Novice Castles place.";
	emotion 0;
	close;

M_WARP:
	//remove several unallowed buffs
	sc_end SC_ASSUMPTIO;
	sc_end SC_IMPOSITIO;
	sc_end SC_SUFFRAGIUM;
	sc_end SC_MAGNIFICAT;
	sc_end SC_WEAPONPERFECTION;
	sc_end SC_GOSPEL;
	sc_end SC_BASILICA;
	sc_end SC_MAGICPOWER;
	sc_end SC_MARIONETTE;
	sc_end SC_MARIONETTE2;
	sc_end SC_DEVOTION;
	sc_end SC_SACRIFICE;
	sc_end SC_MAXOVERTHRUST;
	sc_end SC_SPIRIT;
	warp "n_castle",102,93+rand(14);
	close;
}


n_castle,102,107,5	script	Cita	729,{
	mes "[Cita]";
	mes "Hello, "+ strcharinfo(0) +". Can I help you?";
	next;
	menu "Warp me to Prontera!",-,"Cancel",LEnd;
	
		warp "prontera",155,177+rand(5);
		close;
	LEnd:
		mes "[Cita]";
		mes "Ok.";
		close;
}
