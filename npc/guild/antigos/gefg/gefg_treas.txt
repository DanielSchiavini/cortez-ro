// Treasure Spawn ---------------------------------------------------------
gefg_cas04,1,1,1	script	Treasure_G04	-1,{

OnRecvCastleG04:
	end;
OnInit:
	if($boxNumG04 == 0) end;
	set $@bxG04, $boxNumG04;
	callfunc "F_GldTreas","gefg_cas04","G04",$boxNumG04,$@bxG04,$@boxIdG04,1354,112,114,119,123,1;
	end;

OnDied:
	mapannounce "gefg_cas04","Tesouro Aberto!",17;
	set $boxNumG04, $boxNumG04 -1;
	if($boxNumG04 == 0) mapannounce "gefg_cas04", "Todos os tesouros foram abertos. Você deve esperar até amanhã para que apareçam mais tesouros.",0;
	end;
}
