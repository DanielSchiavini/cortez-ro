function	script	F_GldDunSw	{

	mes "[ Voz Misteriosa ]";
	mes " ' Somente grandes guerreiros podem liderar o castelo.. '";
	next;
	mes "(Há um pequeno botão por aqui....... Deseja apertá-lo?)";
	next;
	menu "Sim",-,"Não",M_No;

		set @GID, GetCastleData(getarg(0)+"",1);
		if (@GID == 0 || getcharid(2) != @GID) goto L_NotGLead;
		warp "gld_dun"+getarg(1)+"",getarg(2),getarg(3);
		end;

		L_NotGLead:
			mes "[ Voz Misteriosa ]";
			mes " ' ..... parece que você não é grande o suficiente...... ' ";
	M_No:
		return;
}
